## Monthly Hotel Booking Analysis Based on Hotel Type

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_csv('hotel_bookings_data.csv')
df.head()


df.loc[df['is_canceled'] == 0, 'not_canceled_bookings'] = 1
sns.barplot(x="not_canceled_bookings", y="arrival_date_month", data=df, estimator=sum, hue="hotel")
